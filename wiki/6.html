获取文件CRC32值

Category: 代码片段 | Tags: c语言, crc32 | Source: Markdown ----------> Back to Wiki

这是一段获取文件CRC32值的代码，与WinRAR内CRC32值相同。

#include <stdio.h>
#include <stdint.h>
#include <string.h>

#define CRC32_POLY 0xEDB88320U
#define BUFF_SIZE 0x00100000U

typedef struct crc32_handle {
    uint32_t table[256];
} CRC32_Handle;

static uint32_t crc32_update(const CRC32_Handle *handle, const uint8_t *buff, size_t len, uint32_t init) {
    uint32_t crc32 = init ^ 0xFFFFFFFFU;

    while (len--)
        crc32 = (crc32 >> 8) ^ handle->table[(crc32 & 0x000000FFU) ^ *buff++];

    return crc32 ^ 0xFFFFFFFFU;
}

static void crc32_init(CRC32_Handle *handle) {
    if (handle == NULL) return;

    int i = 0, j = 0;
    for (i = 0; i < 256; i++) {
        handle->table[i] = i;
        for (j = 0; j < 8; j++) {
            if (handle->table[i] & 0x00000001U)
                handle->table[i] = (handle->table[i] >> 1) ^ CRC32_POLY;
            else
                handle->table[i] = handle->table[i] >> 1;
        }
    }
}

static uint32_t crc32_file(const CRC32_Handle *handle, const char *file) {
    uint32_t crc32 = 0;
    uint8_t buff[BUFF_SIZE] = {0};
    size_t length = 0;

    FILE *fd = fopen(file, "rb");
    if (fd == NULL) return 0;
    while ((length = fread(buff, sizeof(uint8_t), BUFF_SIZE, fd)) > 0)
        crc32 = crc32_update(handle, buff, length, crc32);
    fclose(fd);

    return crc32;
}

static uint32_t crc32_buff(const CRC32_Handle *handle, const uint8_t *buff, size_t len, const uint32_t init) {
    if (handle == NULL || buff == NULL || len <= 0) return 0;

    return crc32_update(handle, buff, len, init);
}

int main(int argc, char **argv) {
    if (argc != 2) {
        printf("Usage: %s <file>\n", argv[0]);
        return 1;
    }

    CRC32_Handle handle;
    crc32_init(&handle);
    printf("file: %08x\n", crc32_file(&handle, argv[1]));
    printf("buff: %08x\n", crc32_buff(&handle, argv[1], strlen(argv[1]), 0));

    return 0;
}
